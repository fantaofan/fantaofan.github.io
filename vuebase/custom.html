# 自定义指令

// main.js
// 自定义指令
Vue.directive("has", {
  inserted(el, binding, vnode) {
    const { value } = binding;
    // 从vuex里面找到按钮的true/false
    let flag = vnode.context.$store.state.route.btnPermisson[value];
    // 如果flag是false，就从当前文档删除
    !flag && el.parentNode.removeChild(el);
  },
});

使用

<button v-has="'edit_button'">编辑</button>
<button v-has="'dele_button'">删除</button>
