# 插槽

根据 type 值，显示不同的标签类型

About.vue
<template>
  <div class="about">
    <Level type="1">1</Level>
    <Level type="2">2</Level>
    <Level type="3">3</Level>
    <Level type="4">4</Level>
    <Level type="5">5</Level>
  </div>
</template>
<script>
import Level from "./Level";
export default {
  components: {
    Level,
  },
};
</script>

Level.vue
<template>
  <div>
    <h1 v-if="type == 1">
      <slot></slot>
    </h1>
    <h2 v-if="type == 2">
      <slot></slot>
    </h2>
    <h3 v-if="type == 3">
      <slot></slot>
    </h3>
    <h4 v-if="type == 4">
      <slot></slot>
    </h4>
    <h5 v-if="type == 5">
      <slot></slot>
    </h5>
  </div>
</template>
<script>
export default {
  props: {
    type: {
      type: String,
    },
  },
};
</script>

具名插槽

About.vue
<template>
  <div class="about">
    <List :arr="arr">
      <!-- 用来接收子组件传递过来的值 -->
      <!-- v-slot 新版本 slot-scope 老版本 -->
      <!-- v-slot:title :title表示插槽的名字 -->
      <!-- # === v-slot: -->
      <template #title="{ item, b }">
        <span>{{ item }}{{ b }}</span>
      </template>
    </List>
  </div>
</template>
<script>
import List from "./List";
export default {
  data() {
    return {
      arr: ["苹果", "柚子", "椰子", "橘子"],
    };
  },
  components: {
    List,
  },
};
</script>

List.vue
<template>
  <div>
    <div v-for="(item, key) in arr" :key="key">
      <!-- 如果不使用插槽，显示li -->
      <slot name="title" :item="item" b="2">
        <li>{{ item }}</li>
      </slot>
    </div>
  </div>
</template>
<script>
export default {
  // 接收父组件about传递的值
  props: ["arr"],
};
</script>
