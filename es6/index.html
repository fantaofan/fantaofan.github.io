# 三、Proxy 和 Reflect

// 一）、proxy和defineProperty的区别
1. proxy能代理数组
2. 兼容性 proxy ie11以上 defineProperty 兼容 ie8以上
3. proxy是代理，defineProperty是劫持
4. 配合Reflect，Reflect有13种方法，处理数据更灵活
// 二）、基本用法
let obj = { name: 'lilei' };
let proxy = new Proxy(obj, {
  // target目标对象，key 目标对象的key
  get(target, key) {
    return target[key];
  },
  //val 需要设置的值
  set(target, key, val) {
    target[key] = val;
  }
});
proxy.name = 1;
// 三）、代理数据 实现更新
let obj = {
  name: "lilei",
  hobit: {
    one: "football",
    two: "eat"
  },
  ary: [1, 2, 3, 4]
};
function update() {
  console.log("更新");
}
handle = {
  // target 目标对象，key 目标对象的key
  get(target, key) {
    // 判断target[key]是不是对象，如果是再返回劫持的对象（继续代理）
    if (typeof target[key] == "object" && typeof target[key] != null) {
      return new Proxy(target[key], handle);
    }
    return Reflect.get(target, key);
    // return target[key];
  },
  // val 需要设置的值
  set(target, key, val) {
    // 因为数组的长度改变也会触发更新
    if (key == "length") return true;
    update();
    return Reflect.set(target, key, val);
    // target[key] = val;
  }
};
let proxy = new Proxy(obj, handle);
console.log(proxy);
proxy.name = "lala";
proxy.hobit.one = "run";
proxy.ary.push(5);
console.log(proxy);
// 数组的更新会触发2次
// 数组触发更新触发两次是为什么
// 检测到length的改变
// 四）、参数使用
let obj = { name: 1 };
let handle = {
  // receiver 它总是指向原始的读操作所在的那个对象，一般情况下就是 Proxy 实例
  get(target, key, receiver) {
    return receiver;
  },
  set(target, key, val, receiver) {
    return Reflect.set(target, key, val, receiver);
  }

let proxy = new Proxy(obj, handle);

# 四、Generator

// 基本写法
function* helloWorldGenerator() {
  // yield 产出
  yield "hello";
  yield "world";
  // 如果return后面没有值等于默认返回了一个undefined
  return "ending";
}
// 调用 Generator 函数后，该函数并不执行，返回的也不是函数运行结果，而是一个指向内部状态的指针对象
let hw = helloWorldGenerator();
// 调用遍历器（hw）的 next 方法，指针指向下一个状态，碰到 yield 或者 return 会暂停
console.log(hw.next());
console.log(hw.next());
console.log(hw.next());
console.log(hw.next());

// Generator 函数具有symbol.iterator接口，所以可以使用 for of 和 ...拓展运算符
function* fn() {
  yield 1;
  yield 2;
  yield 3;
}
console.log([...fn()]); // 直接拿到每一步的执行结果
for (let key of fn()) {
  console.log(key);
}
console.log(Array.from(fn()));

// throw方法抛出的错误要被内部捕获，前提是必须至少执行过一次next方法

// next()、throw()、return()这三个方法本质上是同一件事，可以放在一起理解。它们的作用都是让 Generator 函数恢复执行，并且使用不同的语句替换yield表达式
// throw()、return() throw就等于代码里面加了throw new Error() return就等于代码里面加了一句return xxx
let it = read();
it.next();
// return提前终止
console.log(it.return(1));

// yield 表达式
// 由于 Generator 函数返回的遍历器对象，只有调用next方法才会遍历下一个内部状态，所以其实提供了一种可以暂停执行的函数。yield表达式就是暂停标志。

// 遍历器对象的next方法的运行逻辑如下。

// （1）遇到yield表达式，就暂停执行后面的操作，并将紧跟在yield后面的那个表达式的值，作为返回的对象的value属性值。

// （2）下一次调用next方法时，再继续往下执行，直到遇到下一个yield表达式。

// （3）如果没有再遇到新的yield表达式，就一直运行到函数结束，直到return语句为止，并将return语句后面的表达式的值，作为返回的对象的value属性值。

// （4）如果该函数没有return语句，则返回的对象的value属性值为undefined。

// 回调 -> promise -> generator -> async await
// generator 和 promise
function getProvince() {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      resolve("北京");
    }, 2000);
  });
}
function getCity(province) {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      if (province === "北京") {
        resolve("昌平");
      }
    }, 1000);
  });
}
getProvince().then((data) => {
  getCity(data).then((data) => {
    console.log(data);
  });
});
function* getArea() {
  let province = yield getProvince();
  let city = yield getCity(province);
  return city;
}
// generator 配合 promise的基本使用
let gt = getArea();
let { value } = gt.next();
value.then((data) => {
  let { value } = gt.next(data);
  value.then((data) => {
    let { value } = gt.next(data);
    console.log(value);
  });
});
// co库
function co(it) {
  return new Promise((resolve, reject) => {
    function next(data) {
      // next()
      let { value, done } = it.next(data); // 第一次undefined
      if (!done) {
        // Promise.resolve(value) 将value转成Promise
        Promise.resolve(value).then(
          (data) => {
            next(data); // 1 next
          },
          (err) => {
            it.throw(err);
          }
        );
      } else {
        // 如果done为true，表示generator执行完毕
        resolve(value);
      }
    }
    next();
  });
}
co(getArea())
  .then((data) => {
    console.log(data);
  })
  .catch((err) => {
    console.log(err);
  });

# 七、for of 只能遍历接口有 symbol.iterator 的属性的可迭代对象（包括 Array，Map，Set，String，TypedArray，arguments 对象，nodelist 等等）

// 用法
for (variable of iterable) {
  // statements
}

// 迭代Array
let ary = [1, 2, 3, 4];
// continue break
for (let key of ary) {
  // if (key === 2) continue;
  console.log(key); // 1 2 3 4 返回数组每一项
}

// 迭代 arguments 对象
(function() {
  for (let argument of arguments) {
    console.log(argument); // 1 2 3
  }
})(1, 2, 3);

// 遍历dom节点 给页面所有的p标签增加class
let plist = document.querySelectorAll("p");
for (let p of plist) {
  p.classList.add("abc");
}

// 迭代 String
let iterable = "boo";
for (let value of iterable) {
  console.log(value);
}

// 遍历Set
let iterable = new Set([1, 1, 2, 2, 3, 3]);
for (let value of iterable) {
  console.log(value); // 1 2 3
}

// 遍历Map
let iterable = new Map([
  ["a", 1],
  ["b", 2],
  ["c", 3],
]);
for (let entry of iterable) {
  console.log(entry); // ["a", 1] ["b", 2] ["c", 3]
}
for (let [key, value] of iterable) {
  console.log(value); // 1 2 3
}
