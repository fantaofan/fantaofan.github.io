# 三、判断权限获取，如果没有获取就发送请求获取

  • 默认设置没有获取过权限，在 vuex 里面设置没有获取过权限
state: {
  hasPermisson: false // 初始化设置用户没有获取过权限
},
  • 在路由的钩子函数里面判断有没有获取过权限，如果没有就获取
// 路由钩子
router.beforeEach(async (to, from, next) => {
  // 如果没有获取过权限，则获取路由权限
  if (!store.state.hasPermisson) {
    // 发送请求获取权限
    const newRouter = await store.dispatch("route/getAuthRoute");
  } else {
    next();
  }
});

# 四、在 vuex 中获取权限列表，并过滤出需要的路由

const state = {
  menuList: [],
  btnPermisson: {
    edit_button: true,
    dele_button: true,
  },
};

const actions = {
  async getAuthRoute({ commit }) {
    let needRouter;
    try {
      // getAuth 获取后台返回的权限列表的请求
      const authList = await getAuth();
      const {
        rules: {
          page,
          component: { edit_button, dele_button },
        },
      } = authList;
      // 根据接口设置按钮权限，传多个值用对象
      commit("setComp", { edit_button, dele_button });
      // 根据权限列表来过滤出我们需要的路由列表
      needRouter = getAccesRouterList(routerMap, page);
      // 设置菜单列表数据
      commit("setMenulist", needRouter);
      // 已经获取过权限了，修改获取权限的状态
      commit("setPermisson", null, { root: true }); // 提交到全局
    } catch (error) {
      console.log(error);
    }
    return needRouter;
  },
};

const mutations = {
  setMenulist(state, routes) {
    state.menuList = routes;
  },
  setComp(state, { edit_button, dele_button }) {
    state.btnPermisson.edit_button = edit_button;
    state.btnPermisson.dele_button = dele_button;
  },
};

// 过滤，返回有权限的路由列表
function getAccesRouterList(routes, page) {
  return routes.filter((route) => {
    if (page[route.name]) {
      // 如果有子路由，递归判断子路由权限，判断子路由是否有权限
      if (route.children)
        route.children = getAccesRouterList(route.children, page);
      return true;
    }
    return false;
  });
}

# 五、动态添加路由

router.beforeEach(async (to, from, next) => {
  // 如果没有获取过权限，则获取路由权限
  if (!store.state.hasPermisson) {
    // 发送请求获取权限
    const newRouter = await store.dispatch("route/getAuthRoute");
    // vue-router Api 提供的动态添加路由的方法
    // router.addRoutes动态添加更多的路由规则。参数必须是一个符合 routes 选项要求的数组
    +router.addRoutes(newRouter); // 动态添加路由
    // 保证路由加载完
    +next({ ...to, replace: true }); // hack写法
  } else {
    next();
  }
});

# 六、构建菜单组件（递归组件）

<!-- menuList是从vuex取出来的 MyMenu.vue -->
<template>
  <div style="">
    <!-- <el-menu :router="true"> -->
    <el-menu>
      <template v-for="menu in menuList">
        <!-- 递归组件 -->
        <ReSubMenu :item="menu" :key="menu.name"></ReSubMenu>
      </template>
    </el-menu>
  </div>
</template>
<script>
  import ReSubMenu from "./ReSubMenu";
  import { mapState } from "vuex";
  export default {
    components: { ReSubMenu },
    computed: {
      ...mapState("route", ["menuList"]),
    },
  };
</script>

<!-- 菜单递归组件 -->
<template>
  <!-- 这里写递归组件 -->
  <div>
    <template>
      <!-- 如果有儿子 -->
      <el-submenu
        v-if="item.children && item.children.length"
        :index="item.path"
        :key="item.name"
      >
        <template slot="title">
          <router-link :to="{ path: item.path }">
            {{ item.meta.title }}
          </router-link>
        </template>
        <template v-for="(m, i) in item.children">
          <!-- 组件里面调用自身 -->
          <ReSubMenu :item="m" :key="i"></ReSubMenu>
        </template>
      </el-submenu>
      <!-- 如果没有儿子 -->
      <el-menu-item v-else :index="item.path">
        <!-- {{ item.meta.title }} -->
        <router-link :to="{ path: item.path }">
          {{ item.meta.title }}
        </router-link>
      </el-menu-item>
    </template>
  </div>
</template>
<script>
  export default {
    // 必须起名字
    name: "ReSubMenu", // 组件是可以在它们自己的模板中调用自身的。不过它们只能通过 name 选项来做这件事
    props: ["item"],
  };
</script>

# 七、组件权限（服务端数据获取的，组件如果较多可以使用自定义指令）

// 自定义指令
Vue.directive("has", {
  inserted(el, binding, vnode) {
    const { value } = binding;
    // 从vuex里面找到按钮的true/false
    let flag = vnode.context.$store.state.route.btnPermisson[value];
    // 如果flag是false，就从当前文档删除
    !flag && el.parentNode.removeChild(el);
  }
});
// 用法
<button v-has="'edit_button'">编辑</button>
<button v-has="'dele_button'">删除</button>
