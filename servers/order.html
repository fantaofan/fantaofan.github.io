# 腾讯云服务器常用命令

netstat -antp | grep 8022 查看端口放没放行
firewall-cmd --list-port    查看放行的端口列表
systemctl status firewalld 查看防火墙状态
systemctl stop firewalld   停止防火墙
firewall-cmd --zone=public --add-port=1997/tcp --permanent     这是添加端口的
ls            ******查看文件夹
rm -rf 文件名 *****删除对应文件夹/文件
i             *****输入
Esc + :wq     *****保存
env           *****查看环境变量
vi 文件名     ***** 打开某个文件
