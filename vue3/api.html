# API

# setup()

代替 vue2 中的 data 和 methods 属性，直接把逻辑写在 setup 里即可

# ref()

一个神奇的函数，要在 template 中使用的变量，必须用 ref 包装 无论赋值还是获取值都需要使用 value

return 出去的数据和方法，在模板中才可以使用，这样可以精准的控制暴露的变量和方法

export default defineComponent({
  name: "App",
  // setup 可以代替vue2中的data和methods属性，直接把逻辑写在setup里即可
  // ref 无论赋值还是获取值都需要使用 value
  // 只要在 template 中使用都必须 return
  setup() {
    const girls = ref(["大脚", "刘英", "晓红"]);
    const selectGirl = ref("");
    // ts 需要定义类型
    const selectGirlFun = (index: number) => {
      selectGirl.value = girls.value[index];
    };
    return {
      girls,
      selectGirl,
      selectGirlFun,
    };
  },
});

# reactive()

是一个函数，接收一个对象

# toRefs()

解决了每次渲染都需要加 data 的问题

// reactive() 接收一个对象
import { defineComponent, ref, reactive, toRefs } from "vue";
// 接口：给data增加类型注解
interface DataProps {
  girls: string[];
  selectGirl: string;
  selectGirlFun: (index: number) => void;
}

export default {
  name: "App",
  setup() {
    // 类型注解
    const data: DataProps = reactive({
      girls: ["大脚", "刘英", "晓红"],
      selectGirl: "",
      selectGirlFun: (index: number) => {
        data.selectGirl = data.girls[index];
      },
    });
    // toRefs
    const refData = toRefs(data);

    return {
      ...refData,
    };
  },
};

# Suspense 组件

提供两个 template 的位置

  • default：异步请求完成后，显示的模板内容
  • fallback：在加载中时，显示的模板内容
  • 注：使用 Suspense 的话，要返回一个 promise 对象，而不是原来的那种 JSON 对象
// GirlShow.vue
<template>
  <img :src="result && result.imgUrl" />
</template>
<script lang="ts">
import axios from "axios";
import { defineComponent } from "vue";
export default defineComponent({
  async setup() {
    const rawData = await axios.get(
      "https://apiblog.jspang.com/default/getGirl"
    );

    return { result: rawData.data };
  },
});
</script>

// AsyncShow.vue
<template>
  <h1>{{ result }}</h1>
</template>
<script lang="ts">
import { defineComponent } from "vue";
export default defineComponent({
  setup() {
    return new Promise((resolve, reject) => {
      setTimeout(() => {
        return resolve({ result: "JSPang" });
      }, 2000);
    });
  },
});
</script>

// App.vue
<template>
  <div>
    <Suspense>
      <template #default>
        <!-- <AsyncShow /> -->
        <girl-show />
      </template>
      <template #fallback>
        <h1>loading...</h1>
      </template>
    </Suspense>
  </div>
</template>
<script lang="ts">
// import AsyncShow from "./components/AsyncShow.vue";
import GirlShow from "./components/GirlShow.vue";
// 捕获异常
import { onErrorCaptured } from "vue";
export default {
  name: "App",
  // components: { AsyncShow },
  components: { GirlShow },
  setup() {
    onErrorCaptured((error) => {
      console.log("error===>", error);
      return true;
    });
    return {};
  },
};
</script>

# Teleport 组件

独立组件，vue2 不支持

  1. 不使用 teleport 遇到的问题：

    • Dialog 被包裹在其它组件之中，容易被干扰
    • 样式也在其它组件中，容易变得非常混乱

  2. teleport 解决的问题：

    • Dialog 组件渲染到任意想渲染的外部 Dom 上，不必嵌套在#app 里，这样互不干扰
// Modal.vue // to 挂载到哪个节点上
<template>
  <teleport to="#modal">
    <div id="center">
      <h2>JSPang.com</h2>
    </div>
  </teleport>
</template>

export default {};
